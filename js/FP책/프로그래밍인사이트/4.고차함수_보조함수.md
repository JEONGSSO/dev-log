```js
   var custom = custom || {}

   custom.utils = function () {
      function rest(iter) {
         var res = [];
         _.forEach(iter, function (v, key) {
            if(key !== 0) res[key - 1] = v
         })
         return res
      }

      return {
         method: function(method) {
            var args = rest(arguments)
            return function(obj) {
               return obj[method].apply(obj, args.concat(rest(arguments)))
            }
         },
      }
   }
```
