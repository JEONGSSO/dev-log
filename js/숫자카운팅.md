function countUp(number) {
  var start = 0;
  var countNum = 300;
  var countEl = document.querySelector('#tt');
  const randomCount = (max, min) => Math.floor(Math.random() * (max - min) + min);
  const countUp = (count) => {
    countEl.innerHTML = count > countNum ? countNum : count;
    const tt = setTimeout(() => {
      start += randomCount(10,17);
      countUp(start);
    },30)
    if (start > countNum) {
      clearTimeout(tt);
    }
  }
  countUp(++start);
}
